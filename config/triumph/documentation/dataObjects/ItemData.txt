These are the functions which can be called on an ItemData object.



setDurabilityMax:
    Arguments:
        float
    Usage:
        max
    Notes:
        Sets the max durability.

setOreDict:
    Arguments:
        String
    Usage:
        ore dict name
    Notes:
        Sets the ore dictionary.

setCountMin:
    Arguments:
        float
    Usage:
        min
    Notes:
        Sets the min stack count.

setEnchantment:
    Arguments:
        String
    Usage:
        enchantment path
    Notes:
        Sets the enchantment.

setDurabilityMin:
    Arguments:
        float
    Usage:
        min
    Notes:
        Sets the min durability.

setPotionType:
    Arguments:
        String
    Usage:
        potion type
    Notes:
        Sets the potion type.

setCountMax:
    Arguments:
        float
    Usage:
        max
    Notes:
        Sets the max stack count.

setCountMinMax:
    Arguments:
        float, float
    Usage:
        min, max
    Notes:
        Sets the min and max stack count.

setItem:
    Arguments:
        String
    Usage:
        item name
    Notes:
        Sets the item. This counts for any meta.

setItem:
    Arguments:
        String, Integer
    Usage:
        item name, meta
    Notes:
        Sets the item with the proided meta.

setEnchantmentLevelsMax:
    Arguments:
        float
    Usage:
        max
    Notes:
        Sets the max enchantment level.

setDurabilityMinMax:
    Arguments:
        float, float
    Usage:
        min, max
    Notes:
        Sets the min and max durability.

setEnchantmentLevelsMinMax:
    Arguments:
        float, float
    Usage:
        min, max
    Notes:
        Sets the min and max enchantment levels.

setEnchantmentLevelsMin:
    Arguments:
        float
    Usage:
        min
    Notes:
        Sets the min enchantment level.
